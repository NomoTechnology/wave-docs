---
title: "List Payments"
api: "GET https://us-central1-jawgames-production-6aaf.cloudfunctions.net/customers/{phoneNumber}/charges"
description: "Lista todos os pagamentos de um cliente."
---

### Response

<ResponseField name="id" type="string">
  Identificador da cobrança.
</ResponseField>

<ResponseField name="amount" type="number">
  Valor em reais.
</ResponseField>

<ResponseField name="catalogProductId" type="string">
  Identificador produto contratado.
</ResponseField>

<ResponseField name="createdAt" type="datetime">
  Data e hora de criação do pagamento.
</ResponseField>

<ResponseField name="customerId" type="string">
  Identificador do cliente.
</ResponseField>

<ResponseField name="lastRetry" type="datetime">
  Data da última tentativa de cobrança.
</ResponseField>

<ResponseField name="paymentProcessorResponseData" type="object">
  Retorno do gateway de pagamento.
</ResponseField>

<ResponseField name="retries" type="number">
  Número de tentativas de pagamento.
</ResponseField>

<ResponseField name="status" type="string">
  Status da cobrança.
</ResponseField>

<ResponseField name="subscriptionId" type="string">
  Identificador da assinatura do cliente.
</ResponseField>
