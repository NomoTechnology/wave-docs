---

### Response

<ResponseField name="amount" type="number">
  Valor em reais.
</ResponseField>

<ResponseField name="catalogProductId" type="string">
  Identificador produto contratado.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Data e hora de criação do pagamento.
</ResponseField>

<ResponseField name="customerId" type="number">
  Identificador do cliente.
</ResponseField>

<ResponseField name="id" type="number">
  Identificador da cobrança.
</ResponseField>

<ResponseField name="lastRetry" type="string">
  Última tentativa de cobrança.
</ResponseField>

<ResponseField name="paymentProcessorResponseData" type="string">
  Retorno do gateway de pagamento.
</ResponseField>

<ResponseField name="retries" type="number">
  Número de tentativas de pagamento.
</ResponseField>

<ResponseField name="status" type="text">
  Status cobrança.
</ResponseField>

<ResponseField name="subscriptionId" type="string">
  Identificador da assinatura do cliente.
</ResponseField>
