api: "GET https://us-central1-jawgames-production-6aaf.cloudfunctions.net/customers/charges"
description: "Este endpoint exibe a lista de cobrança dos clientes."
---

### Response

<ResponseField name="amount" type="number">
  Valor a ser cobrado.
</ResponseField>

<ResponseField name="catalogProductId" type="string">
  Indicates whether a new user group was created.
</ResponseField>

<ResponseField name="createdAt" type="string">
  Data e hora da aquisição.
</ResponseField>

<ResponseField name="customerId" type="number">
  ID do customer.
</ResponseField>

<ResponseField name="id" type="number">
  ID da transação.
</ResponseField>

<ResponseField name="lastRetry" type="string">
  Data e Hora da tentativa de cobrança.
</ResponseField>

<ResponseField name="paymentProcessorResponseData" type="string">
  Indicates whether a new user group was created.
</ResponseField>

<ResponseField name="retries" type="number">
  Número de tentativas.
</ResponseField>

<ResponseField name="status" type="text">
  Status do pagamento do customer.
</ResponseField>

<ResponseField name="subscriptionId" type="string">
  ID de subscription.
</ResponseField>
